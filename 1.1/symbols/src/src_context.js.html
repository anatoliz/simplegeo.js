1 /**
  2  * SimpleGeo Context Client
  3  * @extends simplegeo.Client
  4  * @constructor
  5  */
  6 simplegeo.ContextClient = function(token, options) {
  7   if (!(this instanceof simplegeo.ContextClient)) return new simplegeo.ContextClient(token, options);
  8   simplegeo.Client.call(this, token, options);
  9 }
 10 
 11 simplegeo.ContextClient.prototype = new simplegeo.Client();
 12 
 13 /**
 14  * Retrieve context information for the location
 15  * @param lat
 16  * @param lon
 17  * @param callback
 18  */
 19 simplegeo.ContextClient.prototype.getContext = function(lat, lon, callback) {
 20   var path = "/1.0/context/" + lat + "," + lon + ".json";
 21   return this.request(path, {}, callback);
 22 }
