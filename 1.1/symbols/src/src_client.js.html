1 /**
  2  * This is a the base SimpleGeo Client, which contains common functionality
  3  * that is used by the various product-oriented clients
  4  * @param token a SimpleGeo JSONP token
  5  * @param [options] an Object containing connection options
  6  * @param [options.host="api.simplegeo.com"] the SimpleGeo API host
  7  * @param [options.port="80"] the port of the SimpleGeo API host
  8  * @constructor
  9  */
 10 simplegeo.Client = function(token, options) {
 11     if (!(this instanceof simplegeo.Client)) return new simplegeo.Client(token, options);
 12     options = options || {};
 13     this.token = token;
 14     this.options = options;
 15     this.host = options.host || 'api.simplegeo.com';
 16     this.port = options.port || '80';
 17 
 18     this.apiUrl = 'http://' + this.host + ':' + this.port;
 19 };
 20 
 21 var navigator = window.navigator;
 22 
 23 simplegeo.Client.prototype = {
 24     request: function(path, data, callback) {
 25         data.token = this.token;
 26         data = $.param(data) + '&callback=?';
 27         $.ajax({
 28             url: this.apiUrl + path,
 29             dataType: 'json',
 30             data: data,
 31             success: function(response) {
 32                 if (response.error) {
 33                   var error = new Error(response.message);
 34                   error.code = response.code;
 35                   callback(error);
 36                 } else {
 37                   callback(null, response.data);
 38                 }
 39             },
 40             error: function(xhr, ajaxOptions, err) {
 41                 callback(err);
 42             }
 43         });
 44     },
 45 
 46     /**
 47      * Returns a given feature.
 48      * @param handle A SimpleGeo handle for an object (looks like "SG_...")
 49      * @param callback
 50      */
 51     getFeature: function(handle, callback) {
 52         var path = "/1.0/features/" + handle + ".json";
 53         return this.request(path, {}, callback);
 54     },
 55 
 56     /**
 57      * Returns a list containing the different available categories
 58      * @param callback
 59      */
 60     getFeatureCategories: function(callback) {
 61         var path = "/1.0/features/categories.json";
 62         return this.request(path, {}, callback);
 63     },
 64 
 65     /**
 66      * Use the navigator.geolocation API to get updates about the location.
 67      * If the user denys the applications request for location information,
 68      * and error will be fired to the callback.
 69      * @param [options]
 70      * @param callback
 71      */
 72     watchLocationFromBrowser: function(options, callback) {
 73         var self = this;
 74         if (callback === undefined) {
 75             callback = options;
 76             options = {};
 77         }
 78         if (navigator && navigator.geolocation) {
 79             navigator.geolocation.watchPosition(function(position) {
 80                 position.source = "navigator";
 81                 callback(null, position);
 82             }, function(err) {
 83                 callback(err);
 84             }, options);
 85         } else {
 86             callback(new Error("navigator.geolocation not available"));
 87         }
 88     },
 89 
 90     /**
 91      * Use the navigator.geolocation API to get the location.
 92      * If the user denys the applications request for location information,
 93      * and error will be fired to the callback.
 94      * @param [options]
 95      * @param callback
 96      */
 97     getLocationFromBrowser: function(options, callback) {
 98         var self = this;
 99         if (callback === undefined) {
100             callback = options;
101             options = {};
102         }
103         if (navigator && navigator.geolocation) {
104             navigator.geolocation.getCurrentPosition(function(position) {
105                 position.source = "navigator";
106                 callback(null, position);
107             }, function(err) {
108                 callback(err);
109             }, options);
110         } else {
111             callback(new Error("navigator.geolocation not available"));
112         }
113     },
114 
115     /**
116      * Use the SimpleGeo API to get an estimate of the user's location
117      * by using the IP address of the request.
118      * @param [ipAddress] an IP address to use instead of the request's IP address
119      * @param callback
120      */
121     getLocationFromIP: function(ipAddress, callback) {
122         var path;
123         if (callback === undefined) {
124             callback = ipAddress;
125             path = "/0.1/locate.json";
126         } else {
127             path = "/0.1/locate/" + ipAddress + ".json";
128         }
129         return this.request(path, {}, function(err, data) {
130             if (err) {
131                 callback(err);
132             } else {
133                 var o = {
134                     coords: {
135                       latitude: data.geometry.coordinates[1],
136                       longitude: data.geometry.coordinates[0]
137                     },
138                     source: "simplegeo"
139                 }
140 
141                 callback(null, o);
142             }
143         });
144     },
145 
146     /**
147      * First try get the location from the browser (this will usually ask
148      * the user to approve). If that fails or the user denies the request,
149      * fall back to doing an IP based lookup for the location
150      * @param [options]
151      * @param callback
152      */
153     getLocation: function(options, callback) {
154       var self = this;
155       if (callback === undefined) {
156           callback = options;
157           options = {};
158       }
159       self.getLocationFromBrowser(options, function(err, position) {
160         if (err) {
161           self.getLocationFromIP(callback);
162         } else {
163           callback(null, position);
164         }
165       });
166     }
167 };
