1 /**
  2  * This is a client
  3  * @param token a SimpleGeo JSONP token
  4  * @param [options] an Object containing connection options
  5  * @param [options.host="api.simplegeo.com"] the SimpleGeo API host
  6  * @param [options.port="80"] the port of the SimpleGeo API host
  7  * @constructor
  8  */
  9 var Client = function(token, options) {
 10     if (!(this instanceof Client)) return new Client(token, options);
 11     options = options || {};
 12     this.token = token;
 13     this.options = options;
 14     this.host = options.host || 'api.simplegeo.com';
 15     this.port = options.port || '80';
 16 
 17     this.apiUrl = 'http://' + this.host + ':' + this.port;
 18 };
 19 
 20 var navigator = window.navigator;
 21 
 22 Client.prototype = {
 23     request: function(path, data, callback) {
 24         data.token = this.token;
 25         data = $.param(data) + '&callback=?';
 26         $.ajax({
 27             url: this.apiUrl + path,
 28             dataType: 'json',
 29             data: data,
 30             success: function(response) {
 31                 if (response.error) {
 32                   var error = new Error(response.message);
 33                   error.code = response.code;
 34                   callback(error);
 35                 } else {
 36                   callback(null, response.data);
 37                 }
 38             },
 39             error: function(xhr, ajaxOptions, err) {
 40                 callback(err);
 41             }
 42         });
 43     },
 44 
 45     /**
 46      * Returns a given feature.
 47      * @param handle A SimpleGeo handle for an object (looks like "SG_...")
 48      * @param callback
 49      */
 50     getFeature: function(handle, callback) {
 51         var path = "/1.0/feature/" + handle + ".json";
 52         return this.request(path, {}, callback);
 53     },
 54 
 55     /**
 56      * Use the navigator.geolocation API to get updates about the location.
 57      * If the user denys the applications request for location information,
 58      * and error will be fired to the callback.
 59      * @param [options]
 60      * @param callback
 61      */
 62     watchLocationHTML5: function(options, callback) {
 63         var self = this;
 64         if (callback === undefined) {
 65             callback = options;
 66             options = {};
 67         }
 68         if (navigator && navigator.geolocation) {
 69             navigator.geolocation.watchPosition(function(position) {
 70                 position.source = "navigator";
 71                 callback(null, position);
 72             }, function(err) {
 73                 callback(err);
 74             }, options);
 75         } else {
 76             callback(new Error("navigator.geolocation not available"));
 77         }
 78     },
 79 
 80     /**
 81      * Use the navigator.geolocation API to get the location.
 82      * If the user denys the applications request for location information,
 83      * and error will be fired to the callback.
 84      * @param [options]
 85      * @param callback
 86      */
 87     getLocationHTML5: function(options, callback) {
 88         var self = this;
 89         if (callback === undefined) {
 90             callback = options;
 91             options = {};
 92         }
 93         if (navigator && navigator.geolocation) {
 94             navigator.geolocation.getCurrentPosition(function(position) {
 95                 position.source = "navigator";
 96                 callback(null, position);
 97             }, function(err) {
 98                 callback(err);
 99             }, options);
100         } else {
101             callback(new Error("navigator.geolocation not available"));
102         }
103     },
104 
105     /**
106      * Use the SimpleGeo API to get an estimate of the user's location
107      * by using the IP address of the request.
108      * @param [ipAddress] an IP address to use instead of the request's IP address
109      * @param callback
110      */
111     getLocationIP: function(ipAddress, callback) {
112         var path;
113         if (callback === undefined) {
114             callback = ipAddress;
115             path = "/0.1/locate.json";
116         } else {
117             path = "/0.1/locate/" + ipAddress + ".json";
118         }
119         return this.request(path, {}, function(err, data) {
120             if (err) {
121                 callback(err);
122             } else {
123                 var o = {
124                     coords: {
125                       latitude: data.geometry.coordinates[1],
126                       longitude: data.geometry.coordinates[0]
127                     },
128                     source: "simplegeo"
129                 }
130 
131                 callback(null, o);
132             }
133         });
134    }
135 };
136 
137 simplegeo.Client = Client;
