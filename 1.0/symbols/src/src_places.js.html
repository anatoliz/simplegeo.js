1 /**
  2  * SimpleGeo Places Client
  3  * @extends Client
  4  * @constructor
  5  */
  6 var PlacesClient = function(token, options) {
  7   if (!(this instanceof PlacesClient)) return new PlacesClient(token, options);
  8   simplegeo.Client.call(this, token, options);
  9 }
 10 
 11 PlacesClient.prototype = new simplegeo.Client();
 12 
 13 /**
 14  * Look up a place
 15  * @param handle
 16  * @param callback
 17  */
 18 PlacesClient.prototype.getRecord = function(handle, callback) {
 19   var path = "/1.0/places/" + handle + ".json";
 20   return this.request(path, {}, callback);
 21 };
 22 
 23 /**
 24  * Search for places
 25  * @param lat
 26  * @param lon
 27  * @param options
 28  * @param [options.q]
 29  * @param [options.category]
 30  * @param callback
 31  */
 32 PlacesClient.prototype.search = function(lat, lon, options, callback) {
 33   var path = "/1.0/places/lat,lon.json";
 34   path = path.replace('lat', lat).replace('lon', lon);
 35   return this.request(path, options, callback);
 36 }
 37 
 38 simplegeo.PlacesClient = PlacesClient;
