1 /**
  2  * Create a SimpleGeo Storage Client
  3  * @see simplegeo.Client
  4  * @extends simplegeo.Client
  5  * @constructor
  6  */
  7 simplegeo.StorageClient = function(token, options) {
  8   if (!(this instanceof simplegeo.StorageClient)) return new simplegeo.StorageClient(token, options);
  9   simplegeo.Client.call(this, token, options);
 10   this.name = 'StorageClient';
 11 }
 12 
 13 simplegeo.StorageClient.prototype = new simplegeo.Client();

 15 var endpoints = {
 16   record: '/0.1/records/layer/id.json',
 17   records: '/0.1/records/layer/ids.json',
 18   history: '/0.1/records/layer/id/history.json',
 19   nearby: '/0.1/records/layer/nearby/arg.json',
 20   layers: '/0.1/layers.json',
 21   layer: '/0.1/layers/name.json'
 22 }

 24 /**
 25  * Retrieve a record.
 26  * @param {String} layer
 27  * @param {String} id
 28  * @param {Function} callback See {@link callbacks}.
 29  * Example response data:
 30  * <blockquote><pre>{
 31  *   "id": "1",
 32  *   "geometry": {
 33  *     "coordinates": [
 34  *       -105.27742,
 35  *       40.016950
 36  *     ],
 37  *     "type": "Point"
 38  *   },
 39  *   "properties": {
 40  *     "layer": "com.simplegeo.example",
 41  *   },
 42  *   "type": "Feature",
 43  *   "created": 1275929679,
 44  *   "layerLink": {
 45  *     "href": "http://api.simplegeo.com/0.1/layer/com.simplegeo.example.json"
 46  *   },
 47  *   "selfLink": {
 48  *     "href": "http://api.simplegeo.com/0.1/records/com.simplegeo.example/1.json"
 49  *   }
 50  * }</pre></blockquote>
 51  */
 52 simplegeo.StorageClient.prototype.getRecord = function(layer, id, callback) {
 53     path = endpoints.record;
 54     path = path.replace('layer', layer).replace('id', id);
 55     return this.request(path, {}, callback);
 56 };

128 /**
129  * Query for records near the location of an IP address
130  * @param {String} layer
131  * @param {String} [ip] Use this IP as the lookup.
132  *        Defaults to the IP address of the request if not specified
133  * @param {Object} [options] See <a href='http://simplegeo.com/docs/api-endpoints/simplegeo-storage#nearby'>SimpleGeo Storage documentation</a>
134  * @param {Function} callback See {@link callbacks}.
135  * Example response data:
136  * <blockquote><pre>{
137  *   "features":[...],
138  *   "type": "FeatureCollection",
139  *   "next_cursor": "eyJpZCI6IjE="
140  * }</pre></blockquote>
141  */
142 simplegeo.StorageClient.prototype.getNearbyFromIP = function(layer, ip, options, callback) {
143     if (arguments.length === 3) {
144       // Check if the ip or the options were omitted
145       callback = options;
146       if (ip.trim) {
147         options = {};
148       } else {
149         options = ip;
150         ip = 'ip';
151       }
152     } else if (arguments.length === 2) {
153       callback = ip;
154       ip = 'ip';
155       options = {};
156     }
157     path = endpoints.nearby;
158     path = path.replace('layer', layer).replace('arg', ip);
159     return this.request(path, options, callback);
160 };

173  * }</pre></blockquote>
174  */
175 simplegeo.StorageClient.prototype.getNearbyFromAddress = function(layer, address, options, callback) {
176     if (callback === undefined) {
177       callback = options;
178       options = {};
179     }
180     path = endpoints.nearby;
181     path = path.replace('layer', layer).replace('arg', 'address');
182     options.address = address
183     return this.request(path, options, callback);
184 }
