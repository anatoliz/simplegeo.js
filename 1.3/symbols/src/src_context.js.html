1 /**
  2  * Create a SimpleGeo Context Client
  3  * @see simplegeo.Client
  4  * @extends simplegeo.Client
  5  * @constructor
  6  */
  7 simplegeo.ContextClient = function(token, options) {
  8   if (!(this instanceof simplegeo.ContextClient)) return new simplegeo.ContextClient(token, options);
  9   simplegeo.Client.call(this, token, options);
 10   this.name = 'ContextClient';
 11 }
 12 
 13 simplegeo.ContextClient.prototype = new simplegeo.Client();
 14 
 15 /**
 16  * Retrieve context information for the location.
 17  * @param {Number} lat
 18  * @param {Number} lon
 19  * @param {Function} callback See {@link callbacks}.
 20  * Example response data:
 21  * <blockquote><pre>{
 22  *   query: {
 23  *     latitude: 40.01695,
 24  *     longitude: -105.27742
 25  *   },
 26  *   timestamp: 1292958383.954,
 27  *   features: [...],
 28  *   demographics: {metro_score: 9}
 29  * }</pre></blockquote>
 30  *
 31  */
 32 simplegeo.ContextClient.prototype.getContext = function(lat, lon, callback) {
 33   var path = "/1.0/context/" + lat + "," + lon + ".json";
 34   return this.request(path, {}, callback);
 35 }
 36 
 37 /**
 38  * Retrieve context information for the location of an IP address.
 39  * @param {String} [ip] Use this ip as the lookup.
 40  *        Defaults to the IP address of the request if not specified.
 41  * @param {Function} callback See {@link callbacks}.
 42  * Example response data:
 43  * <blockquote><pre>{
 44  *   query: {
 45  *     latitude: 37.4192,
 46  *     longitude: -122.0574,
 47  *     ip: "74.125.224.16"
 48  *   },
 49  *   timestamp: 1292958831.291,
 50  *   features: [...],
 51  *   demographics: {metro_score: 10}
 52  * }</pre></blockquote>
 53  */
 54 simplegeo.ContextClient.prototype.getContextFromIP = function(ip, callback) {
 55   if (callback === undefined) {
 56     callback = ip;
 57     ip = 'ip';
 58   }
 59   var path = "/1.0/context/" + ip + ".json";
 60   return this.request(path, {}, callback);
 61 }
 62 
 63 /**
 64  * Retrieve context information for an address
 65  * @param {String} address a mailing address to use as the point for the query.
 66  * @param {Function} callback See {@link callbacks}.
 67  * Example response data:
 68  * <blockquote><pre>{
 69  *   query: {
 70  *     latitude: 37.772555,
 71  *     longitude: -122.405978,
 72  *     address: "41 Decatur Street, San Francisco, CA"
 73  *   },
 74  *   timestamp: 1292959054.122,
 75  *   features: [...],
 76  *   demographics: {metro_score: 10}
 77  * }</pre></blockquote>
 78  */
 79 simplegeo.ContextClient.prototype.getContextFromAddress = function(address, callback) {
 80   var path = "/1.0/context/address.json";
 81   return this.request(path, {address: address}, callback);
 82 }
