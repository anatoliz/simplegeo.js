  2   /**
  3    * This is a the base SimpleGeo Client, which contains common functionality
  4    * that is used by the various product-oriented clients
  5    * @param token a SimpleGeo JSONP token
  6    * @param [options] an Object containing connection options
  7    * @param [options.host="api.simplegeo.com"] the SimpleGeo API host
  8    * @param [options.port="80"] the port of the SimpleGeo API host
  9    * @constructor
 10    */
 11   simplegeo.Client = function(token, options) {
 12       if (!(this instanceof simplegeo.Client)) return new simplegeo.Client(token, options);
 13       options = options || {};
 14       this.token = token;
 15       this.options = options;
 16       this.proto = options.proto || 'https';
 17       this.host = options.host || 'api.simplegeo.com';
 18       this.port = options.port || (this.proto === 'https' ? '443' : '80');
 19 
 20       var jqueryVersion = parseFloat($.fn.jquery.match(/^[0-9]+\.[0-9]+/)[0]),
 21           jqueryMin = $.fn.jquerymin;
 22       if (options.cors && jqueryMin) {
 23         throw new Error("You cannot use CORS unless you use the jQuery version of the SimpleGeo client");
 24       }
 25       this.cors = options.cors || ((!jqueryMin && jqueryVersion >= 1.4) ? 'auto' : false);
 26 
 27       this.apiUrl = this.proto + '://' + this.host + ':' + this.port;
 28       this.name = 'Client';
 29   };

 33   simplegeo.Client.prototype = {
 34       request: function(path, data, callback) {
 35         if (this.cors) {
 36           this.requestCORS(path, data, callback);
 37         } else {
 38           this.requestJSONP(path, data, callback);
 39         }
 40       },
 41 
 42       requestCORS: function(path, data, callback) {
 43           var self = this;
 44           data.token = this.token;
 45           try {
 46             $.ajax({
 47                 url: this.apiUrl + path,
 48                 dataType: 'json',
 49                 data: data,
 50                 traditional: true,
 51                 success: function(response) {
 52                     callback(null, response);
 53                 },
 54                 error: function(xhr, textStatus, err) {
 55                     // err will be undefined if this is an HTTP error
 56                     if (!err) {
 57                       var err = {
 58                         error: true,
 59                         code: xhr.status
 60                       }
 61 
 62                       try {
 63                         err.message = JSON.parse(responseText).message;
 64                       } catch (e) {
 65                         err.message = xhr.statusText;
 66                       }
 67 
 68                       callback(err);
 69                     } else {
 70                       if (window.console) console.error(path, xhr, textStatus, err);
 71                       if (self.cors === 'auto') {
 72                         // Fall back to JSONP if CORS fails
 73                         self.cors = false;
 74                         self.requestJSONP(path, data, callback);
 75                       } else {
 76                         callback(err);
 77                       }
 78                     }
 79                 }
 80             });
 81           } catch (err) {
 82             if (window.console) console.error(path, xhr, textStatus, err);
 83             if (self.cors === 'auto') {
 84               // Fall back to JSONP if CORS fails
 85               self.cors = false;
 86               self.requestJSONP(path, data, callback);
 87             } else {
 88               throw err;
 89             }
 90           }
 91       },

 93       requestJSONP: function(path, data, callback) {
 94           data.token = this.token;
 95           data = $.param(data, true) + '&callback=?';
 96           $.ajax({
 97               url: this.apiUrl + path,
 98               dataType: 'json',
 99               data: data,
100               success: function(response) {
101                   if (response.error) {
102                     var error = new Error(response.message);
103                     error.code = response.code;
104                     callback(error);
105                   } else {
106                     callback(null, response.data);
107                   }
108               },
109               error: function(xhr, ajaxOptions, err) {
110                   callback(err);
111               }
112           });
113       },

154       /**
155        * Use the navigator.geolocation API to get updates about the location.
156        * If the user denys the applications request for location information,
157        * and error will be fired to the callback.
158        * @param [options] <a href='http://dev.w3.org/geo/api/spec-source.html#position_options_interface'>W3C Geolocation PositionOptions</a>
159        * @param callback See {@link callbacks}.
160        * The data passed to the callback matches the <a href='http://dev.w3.org/geo/api/spec-source.html#position_interface'>W3C Geolocation Position Interface</a>.
161        */
162       watchLocationFromBrowser: function(options, callback) {
163           var self = this;
164           if (callback === undefined) {
165               callback = options;
166               options = {};
167           }
168           if (navigator && navigator.geolocation) {
169               navigator.geolocation.watchPosition(function(position) {
170                   callback(null, position);
171               }, function(err) {
172                   callback(err);
173               }, options);
174           } else {
175               callback(new Error("navigator.geolocation not available"));
176           }
177       },
178 
179       /**
180        * Use the navigator.geolocation API to get the location.
181        * If the user denys the applications request for location information,
182        * and error will be fired to the callback.
183        * @param [options] <a href='http://dev.w3.org/geo/api/spec-source.html#position_options_interface'>W3C Geolocation PositionOptions</a>
184        * @param callback See {@link callbacks}.
185        * The data passed to the callback matches the <a href='http://dev.w3.org/geo/api/spec-source.html#position_interface'>W3C Geolocation Position Interface</a>.
186        */
187       getLocationFromBrowser: function(options, callback) {
188           var self = this;
189           if (callback === undefined) {
190               callback = options;
191               options = {};
192           }
193           if (navigator && navigator.geolocation) {
194               navigator.geolocation.getCurrentPosition(function(position) {
195                   callback(null, position);
196               }, function(err) {
197                   callback(err);
198               }, options);
199           } else {
200               callback(new Error("navigator.geolocation not available"));
201           }
202       },

204       /**
205        * Use the SimpleGeo API to get an estimate of the user's location
206        * by using the IP address of the request.
207        * @param [ipAddress] an IP address to use instead of the request's IP address
208        * @param callback See {@link callbacks}.
209        * The data passed to the callback matches the <a href='http://dev.w3.org/geo/api/spec-source.html#position_interface'>W3C Geolocation Position Interface</a>.
210        */
211       getLocationFromIP: function(ipAddress, callback) {
212           var path;
213           if (callback === undefined) {
214               callback = ipAddress;
215               path = "/0.1/locate.json";
216           } else {
217               path = "/0.1/locate/" + ipAddress + ".json";
218           }
219           return this.request(path, {}, function(err, data) {
220               if (err) {
221                   callback(err);
222               } else {
223                   var o = {
224                       coords: {
225                         latitude: data.geometry.coordinates[1],
226                         longitude: data.geometry.coordinates[0],
227                         accuracy: 100000
228                       },
229                       timestamp: new Date().getTime(),
230                       source: "simplegeo"
231                   }
232 
233                   callback(null, o);
234               }
235           });
236       },

238       /**
239        * First try get the location from the browser (this will usually ask
240        * the user to approve). If that fails or the user denies the request,
241        * fall back to doing an IP based lookup for the location
242        * @param [options] <a href='http://dev.w3.org/geo/api/spec-source.html#position_options_interface'>W3C Geolocation PositionOptions</a>
243        * @param callback See {@link callbacks}.
244        * The data passed to the callback matches the <a href='http://dev.w3.org/geo/api/spec-source.html#position_interface'>W3C Geolocation Position Interface</a>.
245        */
246       getLocation: function(options, callback) {
247         var self = this;
248         if (callback === undefined) {
249             callback = options;
250             options = {};
251         }
252         self.getLocationFromBrowser(options, function(err, position) {
253           if (err) {
254             self.getLocationFromIP(callback);
255           } else {
256             callback(null, position);
257           }
258         });
259       }
