1 /**
  2  * Create a SimpleGeo Context Client
  3  * @see simplegeo.Client
  4  * @extends simplegeo.Client
  5  * @constructor
  6  */
  7 simplegeo.ContextClient = function(token, options) {
  8   if (!(this instanceof simplegeo.ContextClient)) return new simplegeo.ContextClient(token, options);
  9   simplegeo.Client.call(this, token, options);
 10   this.name = 'ContextClient';
 11 }
 12 
 13 simplegeo.ContextClient.prototype = new simplegeo.Client();
 14 
 15 /**
 16  * Retrieve table numbers for demographics data.
 17  * @param {String} q Search query
 18  * @param {Function} callback See {@link callbacks}.
 19  * Example response data:
 20  * <blockquote><pre>[
 21  * ["B06010", 
 22  *   [ "PLACE OF BIRTH BY INDIVIDUAL INCOME IN THE PAST 12 MONTHS (IN 2009 INFLATION-ADJUSTED DOLLARS) IN THE UNITED STATES", 
 23  *     "Universe:  Population 15 years and over in the United States"],
 24  *   [
 25  *     ["No income ", "B06010035"],
 26  *     ["With income:", "B06010036"],
 27  *     ...
 28  *     ["No income ", "B06010002"]
 29  *   ]
 30  * ],...]</pre></blockquote>
 31  *
 32  */
 33 simplegeo.ContextClient.prototype.searchDemographicTables = function(q, callback) {
 34   var path = "/1.0/context/demographics/search.json";
 35   return this.request(path, {q: q}, callback);
 36 }

 38 /**
 39  * Retrieve context information for the location.
 40  * @param {Number} lat
 41  * @param {Number} lon
 42  * @param {Function} callback See {@link callbacks}.
 43  * Example response data:
 44  * <blockquote><pre>{
 45  *   query: {
 46  *     latitude: 40.01695,
 47  *     longitude: -105.27742
 48  *   },
 49  *   timestamp: 1292958383.954,
 50  *   features: [...],
 51  *   demographics: {metro_score: 9}
 52  * }</pre></blockquote>
 53  *
 54  */
 55 simplegeo.ContextClient.prototype.getContext = function(lat, lon, callback) {
 56   var path = "/1.0/context/" + lat + "," + lon + ".json";
 57   return this.request(path, {}, callback);
 58 }
 59 
 60 /**
 61  * Retrieve context information for the location of an IP address.
 62  * @param {String} [ip] Use this ip as the lookup.
 63  *        Defaults to the IP address of the request if not specified.
 64  * @param {Function} callback See {@link callbacks}.
 65  * Example response data:
 66  * <blockquote><pre>{
 67  *   query: {
 68  *     latitude: 37.4192,
 69  *     longitude: -122.0574,
 70  *     ip: "74.125.224.16"
 71  *   },
 72  *   timestamp: 1292958831.291,
 73  *   features: [...],
 74  *   demographics: {metro_score: 10}
 75  * }</pre></blockquote>
 76  */
 77 simplegeo.ContextClient.prototype.getContextFromIP = function(ip, callback) {
 78   if (callback === undefined) {
 79     callback = ip;
 80     ip = 'ip';
 81   }
 82   var path = "/1.0/context/" + ip + ".json";
 83   return this.request(path, {}, callback);
 84 }
 85 
 86 /**
 87  * Retrieve context information for an address
 88  * @param {String} address a mailing address to use as the point for the query.
 89  * @param {Function} callback See {@link callbacks}.
 90  * Example response data:
 91  * <blockquote><pre>{
 92  *   query: {
 93  *     latitude: 37.772555,
 94  *     longitude: -122.405978,
 95  *     address: "41 Decatur Street, San Francisco, CA"
 96  *   },
 97  *   timestamp: 1292959054.122,
 98  *   features: [...],
 99  *   demographics: {metro_score: 10}
100  * }</pre></blockquote>
101  */
102 simplegeo.ContextClient.prototype.getContextFromAddress = function(address, callback) {
103   var path = "/1.0/context/address.json";
104   return this.request(path, {address: address}, callback);
105 }
