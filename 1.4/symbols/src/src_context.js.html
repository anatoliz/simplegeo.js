1 /**
  2  * Create a SimpleGeo Context Client
  3  * @see simplegeo.Client
  4  * @extends simplegeo.Client
  5  * @constructor
  6  */
  7 simplegeo.ContextClient = function(token, options) {
  8   if (!(this instanceof simplegeo.ContextClient)) return new simplegeo.ContextClient(token, options);
  9   simplegeo.Client.call(this, token, options);
 10   this.name = 'ContextClient';
 11 }
 12 
 13 simplegeo.ContextClient.prototype = new simplegeo.Client();
 14 
 15 /**
 16  * Retrieve table numbers for demographics data.
 17  * @param {String} q Search query
 18  * @param {Function} callback See {@link callbacks}.
 19  * Example response data:
 20  * <blockquote><pre>[
 21  * ["B06010", 
 22  *   [ "PLACE OF BIRTH BY INDIVIDUAL INCOME IN THE PAST 12 MONTHS (IN 2009 INFLATION-ADJUSTED DOLLARS) IN THE UNITED STATES", 
 23  *     "Universe:  Population 15 years and over in the United States"],
 24  *   [
 25  *     ["No income ", "B06010035"],
 26  *     ["With income:", "B06010036"],
 27  *     ...
 28  *     ["No income ", "B06010002"]
 29  *   ]
 30  * ],...]</pre></blockquote>
 31  *
 32  */
 33 simplegeo.ContextClient.prototype.searchDemographicTables = function(q, callback) {
 34   var path = "/1.0/context/demographics/search.json";
 35   return this.request(path, {q: q}, callback);
 36 }

 38 /**
 39  * Retrieve context information for the location.
 40  * @param {Number} lat
 41  * @param {Number} lon
 42  * @param [options] See the <a href='https://simplegeo.com/docs/api-endpoints/simplegeo-context#demographics'>SimpleGeo Context documentation</a>
 43  *                  for the available options. Example: <code>{filter: "demographics.acs.B01001"}</code>
 44  * @param {Function} callback See {@link callbacks}.
 45  * Example response data:
 46  * <blockquote><pre>{
 47  *   query: {
 48  *     latitude: 40.01695,
 49  *     longitude: -105.27742
 50  *   },
 51  *   timestamp: 1292958383.954,
 52  *   features: [...],
 53  *   demographics: {metro_score: 9}
 54  * }</pre></blockquote>
 55  *
 56  */
 57 simplegeo.ContextClient.prototype.getContext = function(lat, lon, options, callback) {
 58   if (callback === undefined) {
 59     callback = options;
 60     options = {};
 61   }
 62   var path = "/1.0/context/" + lat + "," + lon + ".json";
 63   return this.request(path, options, callback);
 64 }

 66 /**
 67  * Retrieve context information for the location of an IP address.
 68  * @param {String} [ip] Use this ip as the lookup.
 69  *        Defaults to the IP address of the request if not specified.
 70  * @param [options] See the <a href='https://simplegeo.com/docs/api-endpoints/simplegeo-context#demographics'>SimpleGeo Context documentation</a>
 71  *                  for the available options. Example: <code>{filter: "demographics.acs.B01001"}</code>
 72  * @param {Function} callback See {@link callbacks}.
 73  * Example response data:
 74  * <blockquote><pre>{
 75  *   query: {
 76  *     latitude: 37.4192,
 77  *     longitude: -122.0574,
 78  *     ip: "74.125.224.16"
 79  *   },
 80  *   timestamp: 1292958831.291,
 81  *   features: [...],
 82  *   demographics: {metro_score: 10}
 83  * }</pre></blockquote>
 84  */
 85 simplegeo.ContextClient.prototype.getContextFromIP = function(ip, options, callback) {
 86   if (callback === undefined) {
 87     if (options === undefined) {
 88       callback = ip;
 89       ip = 'ip';
 90     } else {
 91       callback = options;
 92     }
 93     options = {};
 94   }
 95   var path = "/1.0/context/" + ip + ".json";
 96   return this.request(path, options, callback);
 97 }

 99 /**
100  * Retrieve context information for an address
101  * @param {String} address a mailing address to use as the point for the query.
102  * @param [options] See the <a href='https://simplegeo.com/docs/api-endpoints/simplegeo-context#demographics'>SimpleGeo Context documentation</a>
103  *                  for the available options. Example: <code>{filter: "demographics.acs.B01001"}</code>
104  * @param {Function} callback See {@link callbacks}.
105  * Example response data:
106  * <blockquote><pre>{
107  *   query: {
108  *     latitude: 37.772555,
109  *     longitude: -122.405978,
110  *     address: "41 Decatur Street, San Francisco, CA"
111  *   },
112  *   timestamp: 1292959054.122,
113  *   features: [...],
114  *   demographics: {metro_score: 10}
115  * }</pre></blockquote>
116  */
117 simplegeo.ContextClient.prototype.getContextFromAddress = function(address, options, callback) {
118   var path = "/1.0/context/address.json";
119   if (callback === undefined) {
120     callback = options;
121     options = {};
122   }
123   options["address"] = address;
124   return this.request(path, options, callback);
125 }
