1 /**
  2  * Create a SimpleGeo Places Client
  3  * @see simplegeo.Client
  4  * @extends simplegeo.Client
  5  * @constructor
  6  */
  7 simplegeo.PlacesClient = function(token, options) {
  8   if (!(this instanceof simplegeo.PlacesClient)) return new simplegeo.PlacesClient(token, options);
  9   simplegeo.Client.call(this, token, options);
 10   this.name = 'PlacesClient';
 11   this.placesVersion = '/1.0'
 12 }
 13 
 14 simplegeo.PlacesClient.prototype = new simplegeo.Client();
 15 
 16 /**
 17  * Search for places
 18  * @param lat
 19  * @param lon
 20  * @param [options] See the <a href='http://simplegeo.com/docs/api-endpoints/simplegeo-places#search'>SimpleGeo Places documentation</a>
 21  *                  for the available options. Example: <code>{q: "Starbucks", radius: 10}</code>
 22  * @param callback See {@link callbacks}. Example response data:
 23  * <blockquote><pre>{
 24  *   total: 25,
 25  *   type: "FeatureCollection",
 26  *   features: [...]
 27  * }</pre></blockquote>
 28  */
 29 simplegeo.PlacesClient.prototype.search = function(lat, lon, options, callback) {
 30   if (callback === undefined) {
 31     callback = options;
 32     options = {};
 33   }
 34   var path = this.placesVersion + "/places/lat,lon.json";
 35   path = path.replace('lat', lat).replace('lon', lon);
 36   return this.request(path, options, callback);
 37 }

 39 /**
 40  * Search for places near the location of an IP address
 41  * @param [ip] Use this IP as the lookup.
 42  *        Defaults to the IP address of the request if not specified
 43  * @param [options]
 44  * @param callback See {@link simplegeo.PlacesClient#search} for examples of
 45  *                 <code>options</code> and <code>callback</code>.
 46  */
 47 simplegeo.PlacesClient.prototype.searchFromIP = function(ip, options, callback) {
 48   if (arguments.length === 2) {
 49     // Check if the ip or the options were omitted 
 50     callback = options;
 51     if (ip.trim) {
 52       options = {};
 53     } else {
 54       options = ip;
 55       ip = 'ip';
 56     }
 57   } else if (arguments.length === 1) {
 58     callback = ip;
 59     ip = 'ip';
 60     options = {};
 61   }
 62   var path = this.placesVersion + "/places/" + ip + ".json";
 63   return this.request(path, options, callback);
 64 }

 66 /**
 67  * Search for places near an address
 68  * @param address a mailing address to use as the point for the query.
 69  * @param [options]
 70  * @param callback See {@link simplegeo.PlacesClient#search} for examples of
 71  *                 <code>options</code> and <code>callback</code>.
 72  */
 73 simplegeo.PlacesClient.prototype.searchFromAddress = function(address, options, callback) {
 74   if (callback === undefined) {
 75     callback = options;
 76     options = {};
 77   }
 78   var path = this.placesVersion + "/places/address.json";
 79   options.address = address;
 80   return this.request(path, options, callback);
 81 }

100 /**
101  * Search for places
102  * @param swLat
103  * @param swLon
104  * @param neLat
105  * @param neLon
106  * @param [options] See the <a href='http://simplegeo.com/docs/api-endpoints/simplegeo-places#search'>SimpleGeo Places documentation</a>
107  *                  for the available options. Example: <code>{q: "Starbucks", radius: 10}</code>
108  * @param callback See {@link callbacks}. Example response data:
109  * <blockquote><pre>{
110  *   total: 25,
111  *   type: "FeatureCollection",
112  *   features: [...]
113  * }</pre></blockquote>
114  */
115 simplegeo.Places12Client.prototype.searchFromBBox = function(swLat, swLon, neLat, neLon, options, callback) {
116   if (callback === undefined) {
117     callback = options;
118     options = {};
119   }
120   var path = this.placesVersion + "/places/swlat,swlon,nelat,nelon.json";
121   path = path.replace('swlat', swLat).replace('swlon', swLon).replace('nelat', neLat).replace('nelon', neLon)
122   return this.request(path, options, callback);
123 }
